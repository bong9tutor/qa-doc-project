// Host 세션 생성
void UAO_OnlineSessionSubsystem::CreateSession(
    int32 NumPublicConnections, FString MatchType)
{
    SessionSettings->NumPublicConnections = NumPublicConnections;
    SessionSettings->Set(SETTING_MATCHTYPE, MatchType,
        EOnlineDataAdvertisementType::ViaOnlineServiceAndPing);
    SessionInterface->CreateSession(
        *UserId, NAME_GameSession, *SessionSettings);
}
2
세션 매칭
Session Matching — Steam OSS
Steam 백엔드를 통한 세션 검색 및 참가 처리.
▶
UAO_OnlineSessionSubsystem IOnlineSessionPtr
FindSessions → SessionResults 콜백 → JoinSession 순서로 진행됩니다.
// 세션 탐색
void UAO_OnlineSessionSubsystem::FindSessions(int32 MaxSearchResults)
{
    LastSessionSearch = MakeShareable(new FOnlineSessionSearch());
    LastSessionSearch->MaxSearchResults = MaxSearchResults;
    LastSessionSearch->bIsLanQuery = false;
    LastSessionSearch->QuerySettings.Set(
        SEARCH_PRESENCE, true, EOnlineComparisonOp::Equals);
    SessionInterface->FindSessions(*UserId, LastSessionSearch.ToSharedRef());
}

// 로비 — 플레이어 입장 처리
void AAO_GameMode_Lobby::PostLogin(APlayerController* NewPlayer)
{
    Super::PostLogin(NewPlayer);
    if (GetNumPlayers() >= 2) { NotifyLobbyReady(); }
}

레벨 로드
Level Load — ServerTravel
Host가 ServerTravel로 모든 클라이언트를 인게임 맵으로 이동.
▶
ServerTravel SeamlessTravel TransitionMap
bUseSeamlessTravel = true 설정 필수. 미설정 시 모든 PlayerController가 파괴됩니다.
// Host → 게임 레벨로 이동
void AAO_GameMode_Lobby::StartGame()
{
    bUseSeamlessTravel = true;
    GetWorld()->ServerTravel(
        TEXT("/Game/Maps/LV_Meadow_Main?listen"));
}

// 승리 조건 체크
void AAO_GameMode_InGame::CheckWinCondition()
{
    if (PuzzleChecker->AreAllPuzzlesSolved() &&
        FuelManager->IsRequiredFuelCollected())
    {
        MulticastOnVictory();
        TravelToRestRoom();
    }
}
