// AO_GameInstance: 스테이지 인덱스 증가
bool UAO_GameInstance::TryAdvanceStageIndex()
{
    const int32 StageCount = AO_MapRoutes::GetStageCount();  // 3
    if (CurrentStageIndex + 1 < StageCount)
    {
        ++CurrentStageIndex;
        return true;
    }
    return false;
}

// AO_GameMode_Rest: 다음 스테이지 또는 로비로 이동
void AAO_GameMode_Rest::HandleRestExitRequest(AController* Requester)
{
    if (!AO_GI->TryAdvanceStageIndex())
    {
        // 더 이상 스테이지 없음 -> 로비 복귀
        AO_GI->ResetRun();
        RequestSynchronizedServerTravel(LobbyPath);
        return;
    }
    // 다음 스테이지로 이동
    RequestSynchronizedServerTravel(NextStagePath);
}
실패(전멸) 시: AAO_GameMode_Stage::ResetGameAndGoToLobby() → ResetRun() → LV_Lobby로 ServerTravel. 스테이지 인덱스/연료 모두 초기화.

▶
RequestSynchronizedServerTravel HandlePlayerTravel
Voice crash 방지를 위해 직접 World->ServerTravel()을 호출하지 않고, RequestSynchronizedServerTravel()을 사용합니다. 모든 클라이언트가 cleanup 완료를 보고하면 Travel을 시작합니다.
// AO_GameMode_InGameBase.cpp
void AAO_GameMode_InGameBase::RequestSynchronizedServerTravel(const FString& URL)
{
    bIsTravelSyncInProgress = true;
    PendingTravelURL = URL;

    for (FConstPlayerControllerIterator It = World->GetPlayerControllerIterator(); It; ++It)
    {
        // 1) 인벤토리 + 체력 저장
        PS->bIsTraveling = true;
        HandlePlayerTravel(PS);

        // 2) 클라이언트에 정리 요청
        PC->Client_PrepareForTravel(URL);
    }
}

// 모든 클라이언트 정리 완료 시
void AAO_GameMode_InGameBase::NotifyPlayerCleanupCompleteForTravel(...)
{
    if (CleanupCompletePlayers.Num() >= TotalPlayers)
        StartServerTravel();  // World->ServerTravel(PendingTravelURL)
}

// AO_PlayerState.cpp: Seamless Travel 시 데이터 복사
void AAO_PlayerState::CopyProperties(APlayerState* PlayerState)
{
    Super::CopyProperties(PlayerState);
    AAO_PlayerState* PS = Cast<AAO_PlayerState>(PlayerState);

    // 커스터마이징 데이터 (항상 복사)
    PS->CharacterCustomizingData = this->CharacterCustomizingData;

    // 인벤토리 (SafeZone 판정에 따라 복사 또는 삭제)
    if (bInventoryShouldPersist)
        PS->PersistentInventory = this->PersistentInventory;
    else
        PS->PersistentInventory.Empty();

    // DeathCount (로비 출발 시 초기화, 그 외 유지)
    if (CurrentMapName.Contains(LobbyName))
        DeathCount = 0;
    else
        PS->DeathCount = this->DeathCount;

    // 체력 (KH: PersistentHealth)
    PS->bHasPersistentHealth = this->bHasPersistentHealth;
    PS->PersistentHealth = this->PersistentHealth;
}

4. Save/Restore 코드 패턴
Save/Restore Code Patterns
A
인벤토리 Save/Restore 흐름
SaveInventoryBeforeTravel → CopyProperties → BeginPlay Restore
스테이지/휴게소에서 맵 전환 시 인벤토리를 PlayerState에 저장하고, 새 맵에서 복원합니다.
▶
HandlePlayerTravel SaveInventoryBeforeTravel CheckPlayerInsideSafeZone 
// STEP 1: Travel 전 저장 (AO_GameMode_InGameBase)
void AAO_GameMode_InGameBase::HandlePlayerTravel(AAO_PlayerState* PS)
{
    // 인벤토리를 PS에 복사
    UAO_InventoryComponent* Inv = Pawn->FindComponentByClass<UAO_InventoryComponent>();
    PS->SaveInventoryBeforeTravel(Inv);

    // SafeZone 판정: 휴게소는 무조건 통과, 스테이지는 좌표 체크
    if (MapName.Contains(TEXT("Rest")))
        bFinalPersist = true;
    else
        bFinalPersist = CheckPlayerInsideSafeZone(PS);

    PS->bInventoryShouldPersist = bFinalPersist;
}

// STEP 2: CopyProperties 에서 조건부 복사
if (bInventoryShouldPersist)
    PS->PersistentInventory = this->PersistentInventory;
else
    PS->PersistentInventory.Empty();

// STEP 3: PlayerState::SaveInventoryBeforeTravel
void AAO_PlayerState::SaveInventoryBeforeTravel(UAO_InventoryComponent* Inv)
{
    PersistentInventory = Inv->Slots;
}

5. 실패 모드
Failure Modes & Common Pitfalls
!
bUseSeamlessTravel 누락 → PC 소멸
P0 Critical — Non-Seamless Travel Destroys All Controllers
Seamless 플래그 없이 ServerTravel하면 모든 PC/PS가 파괴되어 데이터가 완전히 소실됩니다.
▶
증상: 맵 전환 후 모든 클라이언트가 재접속한 것처럼 동작. 인벤토리, 체력, 커스터마이징 등 모든 PlayerState 데이터 소실.
원인: GameMode 생성자에서 bUseSeamlessTravel = true 미설정.
검증: AO_GameMode_InGameBase 생성자와 AO_GameMode_Lobby 생성자 모두 설정 여부 확인 필요. 
// 정상 코드 (AO_GameMode_InGameBase.cpp)
AAO_GameMode_InGameBase::AAO_GameMode_InGameBase()
{
    bUseSeamlessTravel = true;  // 이것이 빠지면 P0 버그
}
!
Save 누락 → 아이템/체력 손실
P1 High — Missing SaveBeforeTravel Call
HandlePlayerTravel()이 호출되지 않거나, 특정 경로에서 누락되면 아이템이나 체력이 초기화됩니다.
▶
증상: 맵 전환 후 인벤토리가 비어있거나 체력이 최대치로 초기화됨.
원인: RequestSynchronizedServerTravel() 내부에서 HandlePlayerTravel()을 호출하지만, HandleStageExitRequest()에서도 별도로 SaveInventoryBeforeTravel()을 호출함 (이중 호출 구조).
위험: 새로운 Travel 경로를 추가할 때 Save 호출을 잊기 쉬움. 
// HandleStageExitRequest에서의 Save (첫 번째 호출)
for (APlayerState* P : AO_GS->PlayerArray)
{
    AO_PS->SaveInventoryBeforeTravel(InvComp);  // Stage에서 직접 호출
    HandlePlayerTravel(AO_PS);              // 여기서도 내부적으로 Save 호출
}

// RequestSynchronizedServerTravel에서의 Save (두 번째 호출)
PS->bIsTraveling = true;
HandlePlayerTravel(PS);  // 다시 Save!
Stage exit 경로에서는 Save가 이중으로 호출되지만, Rest exit 경로에서는 HandlePlayerTravel()만 호출됩니다. 인벤토리 Save 없이 bIsTraveling만 설정하는 경로가 있는지 QA 확인이 필요합니다.
!
GameInstance 초기화 실패 → 정의되지 않은 상태
P1 High — Stale GameInstance Data
ResetRun()이 호출되지 않으면 이전 판의 데이터가 다음 판에 유입됩니다.
▶
증상: 새 게임 시작 시 이전 판의 StageIndex, 연료, 부활 횟수가 남아있음.
원인: TravelToStage()에서 ResetRun()을 호출하지만, 비정상 경로(crash 복구, disconnect 후 재접속)에서는 ResetRun()이 누락될 수 있음.
관련 코드: 
// 정상 경로: 로비 -> 스테이지 시작
void AAO_GameMode_Lobby::TravelToStage()
{
    AO_GI->ResetRun();  // 스테이지/연료/부활 초기화
    ...
}

// ResetRun 내용
void UAO_GameInstance::ResetRun()
{
    CurrentStageIndex = 0;
    SharedTrainFuel = GetInitialFuelValue();
    SharedReviveCount = InitialSharedReviveCount;
    TeamDeathCount = 0;
    GameStartTime = 0;
    GameEndTime = 0;
}

// 실패 복귀 경로
void AAO_GameMode_Stage::ResetGameAndGoToLobby(...)
{
    AO_GI->ResetRun();  // 여기서도 초기화
    RollbackSessionInGameFlag();
    RequestSynchronizedServerTravel(LobbyPath);
}
QA 체크포인트:
- 마지막 스테이지 클리어 후 SetGameClear() 경로에서는 ResetRun()이 주석 처리되어 있음 (UI 버튼에서 로비 이동 시 호출 필요)
- bIsGameCleared와 bIsStageFailed는 GameState 소속이므로 맵 전환 시 자동 초기화됨 (별도 리셋 불필요)
!
OnRep 콜백 타이밍 이슈
P2 Medium — OnRep Callbacks After Travel
Travel 완료 후 클라이언트에서 OnRep 콜백이 예상과 다른 시점에 발생할 수 있습니다.
▶
증상: Travel 직후 UI가 갱신되지 않거나, 이전 맵의 데이터가 잠시 표시됨.
원인: Seamless Travel 후 새 World에서 Replication이 완전히 수립되기 전까지 OnRep 콜백이 지연됨.
대응: 서버에서는 HasAuthority() 체크 후 OnRep을 수동 호출하는 패턴을 사용함 (예: OnRep_SharedReviveCount(), OnRep_IsGameCleared()). 
// 서버에서의 수동 OnRep 호출 패턴
void AAO_GameState::SetSharedReviveCount(int32 InValue)
{
    SharedReviveCount = NewValue;
    if (HasAuthority())
    {
        OnRep_SharedReviveCount();  // 서버는 OnRep 자동 호출 안됨
    }
}
